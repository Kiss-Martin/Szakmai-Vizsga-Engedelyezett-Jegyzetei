Resource-ok (Adat transzformáció)

Generálás és Példa

php artisan make:resource ResourceName

A Resource segítségével szabályozható a kimenő JSON szerkezete (mezők átnevezése, formázása):

public function toArray(Request $request): array {
    return [
        'identifier'  => $this->id,
        'title_of_something'  => strtoupper($this->title),
        'things'   => ThingResource::collection($this->whenLoaded('things')),
    ];
}

Backend - Next.js

REST végpontok

standard GET

export async function GET() {
  try {
    const data = await prisma.table.findMany({});
    return NextResponse.json(data);
  } catch (error) {
    return NextResponse.json(
      { message: error instanceof Error ? error.message : "Ismeretlen hiba!" },
      { status: 500 }
    );
  }
}

standard GET by Id

export async function GET(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params;
    const data = await prisma.table.findUnique({
      where: { id: Number(id) },
    });
    return NextResponse.json(data);
  } catch (error) {
    return NextResponse.json(
      { message: error instanceof Error ? error.message : "Ismeretlen hiba!" },
      { status: 500 }
    );
  }
}

standard POST

export async function POST(request: Request) {
  try {
    const body = await request.json();
    const newOne = await prisma.table.create({
      data: body,
    });
    return NextResponse.json(newOne, { status: 201 });
  } catch (error) {
    return NextResponse.json(
      { message: error instanceof Error ? error.message : "Ismeretlen hiba!" },
      { status: 500 }
    );
  }
}

standard PATCH

export async function PATCH(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params;
    const changedData = await request.json();
    const updatedDoc = await prisma.table.update({
      where: { id: Number(id) },
      data: changedData,
    });
    return NextResponse.json(updatedDoc, { status: 200 });
  } catch (error) {
    return NextResponse.json(
      { message: error instanceof Error ? error.message : "Ismeretlen hiba!" },
      { status: 500 }
    );
  }
}

standard PUT

export async function PUT(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params;
    const newDocument = await request.json();
    const updatedDoc = await prisma.table.update({
      where: { id: Number(id) },
      data: newDocument,
    });
    return NextResponse.json(updatedDoc, { status: 200 });
  } catch (error) {
    return NextResponse.json(
      { message: error instanceof Error ? error.message : "Ismeretlen hiba!" },
      { status: 500 }
    );
  }
}

standard DELETE

export async function DELETE(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params;
    await prisma.table.delete({ where: { id: Number(id) } });
    return new NextResponse(null, { status: 204 });
  } catch (error) {
    return NextResponse.json(
      { message: error instanceof Error ? error.message : "Ismeretlen hiba!" },
      { status: 500 }
    );
  }
}

prisma schema example

datasource db {
  provider = "mongodb"
  url      = env("DATABASE_URL")
}

model Collection1 {
  id                Int    @id @map("_id")
  unique_something  String @unique
  default_something String @default("default value")

  viewpoints Collection2[]

  @@index([default_something])
  @@map("collection1name")
}

model Collection2 {
  id          Int       @id @map("_id")
  date        DateTime? @default(now()) @db.Date
  relation_id Int

  location Collection1? @relation(fields: [relation_id], references: [id])

  @@map("collection2name")
}
